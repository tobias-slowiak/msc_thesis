\begin{frame}{Why nanomechanical membranes?}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \begin{itemize}
      \item Conventional FTIR reaches sensitivity limits for small sample masses, in this context we look at approximately $0.6-\SI{40}{\nano\gram}$.
    \end{itemize}
    \vspace{0.4cm}
    \column{0.42\textwidth}
    \centering
    \def\emiliePanelWidth{0.75\linewidth}
    \includegraphics[width=\emiliePanelWidth,trim=0 0 525 0,clip]{EMILIE.png}

    \makebox[\linewidth][c]{\hspace*{-0.05cm}\includegraphics[width=\emiliePanelWidth,trim=475 0 0 0,clip]{EMILIE.png}}
  \end{columns}
\end{frame}

\begin{frame}{Measurement principle}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{enumerate}
      \item Sample is deposited on a prestressed $\silnit$ membrane.
      \item IR absorption heats the sample and membrane.
      \item Temperature changes the membrane stress.
      \item Resonance frequencies shift as a function of wavelength.
    \end{enumerate}
    \column{0.46\textwidth}
    \begin{block}{Mechanical information}
      A deposited sample changes the eigenfrequencies through
      \begin{itemize}
        \item added mass,
        \item possible sample prestress,
        \item spatial overlap with each mode shape.
      \end{itemize}
      \textbf{Goal:} build a mechanical FEM model that links deposited mass distributions to measurable resonance frequencies.
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Research questions}
  \begin{block}{Main question}
    Can a purely mechanical FEM simulation predict deposited sample mass from the eigenfrequency response of an EMILIE membrane?
  \end{block}
  \vspace{0.25cm}
  \begin{itemize}
    \item How accurately can the membrane prestress be estimated?
    \item Are unsampled membranes reproduced over different geometries and manufacturers?
    \item Do sampled membranes behave as expected from added mass alone?
    \item Where are the limits imposed by experiment and material behavior?
  \end{itemize}
\end{frame}

\begin{frame}{Simulation model}
  \begin{columns}[T,onlytextwidth]
    \column{0.8\textwidth}
    \begin{itemize}
      \item The membrane is modeled as a two-dimensional prestressed elastic structure.
      \item Bending stiffness is negligible for the considered modes.
      \item Two coupled PDEs:
      \begin{itemize}
        \item static elasticity for displacement and stress redistribution
            \[
              -\nabla\cdot\fatsig_\fatu = \nabla\cdot\fatsig_p,
            \]
        \item wave eigenvalue problem for resonance frequencies
        \[
            \nabla \cdot \left( \fatsig(x,y) \cdot \nabla u \right) 
    = \rho \, \frac{\partial^2 u(x,y,t)}{\partial t^2}.
    \]
      \end{itemize}
      \item The model is implemented in \ngsolve therefore highly parametrizable.
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Validation: Unsampled membranes}
  \begin{columns}[T,onlytextwidth]
    \column{0.38\textwidth}
    \begin{itemize}
      \item Membrane prestress is fitted using the $(1,1)$ mode.
      \item Higher modes are then predicted by the same mechanical model.
      \item Agreement is within experimental error bounds for a broad parameter range.
    \end{itemize}
    \vspace{0.15cm}
    \begin{block}{Interpretation}
      The model is reliable for estimating membrane prestress and reproducing the baseline eigenfrequency spectrum.
    \end{block}

  \end{columns}
\end{frame}

\begin{frame}{Polypropylene: Model error with sample prestress}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \centering
    \includegraphics[scale=\plotTextMatchScale]{sample_prestress_analysis.pdf}
    \column{0.35\textwidth}
    \begin{itemize}
      \item Agreement improves when internal prestress in the PP deposit is introduced.
    \end{itemize}
    \vspace{0.25cm}
    \begin{block}{Consequence}
      Eigenfrequency shifts are not only a mass signal; they can be masked or reversed by sample prestress.
    \end{block}
  \end{columns}
\end{frame}


\begin{frame}{Conclusions}
  \begin{itemize}
    \item A two-dimensional FEM framework for sampled nanomembranes was developed in \ngsolve.
    \item Unsampled membranes are reproduced within experimental uncertainty after prestress fitting.
    \item Mass prediction is feasible for sufficiently large PS samples, roughly above \SI{10}{\nano\gram}.
    \item Mass prediction is limited for small masses and for materials with significant sample prestress.
  \end{itemize}
\end{frame}
